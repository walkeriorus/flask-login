CREATE DATABASE sounds;
CREATE TABLE `usuarios` (
	`user_id` INT(10) NOT NULL AUTO_INCREMENT,
	`user_name` VARCHAR(25) NOT NULL COLLATE 'utf8mb3_general_ci',
	`user_email` VARCHAR(100) NOT NULL COLLATE 'utf8mb3_general_ci',
	`user_pass` VARCHAR(120) NOT NULL COLLATE 'utf8mb3_general_ci',
	`user_role` TINYINT(1) NOT NULL DEFAULT '0',
	PRIMARY KEY (`user_id`) USING BTREE,
	UNIQUE INDEX `user_name` (`user_name`) USING BTREE
)
COLLATE='utf8mb3_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=14
;
CREATE TABLE `productos` (
	`id_producto` INT(5) NOT NULL AUTO_INCREMENT,
	`nombre` VARCHAR(50) NOT NULL COLLATE 'utf8mb3_general_ci',
	`precio` FLOAT NOT NULL,
	`imagen` VARCHAR(150) NULL DEFAULT NULL COLLATE 'utf8mb3_general_ci',
	PRIMARY KEY (`id_producto`) USING BTREE,
	UNIQUE INDEX `nombre` (`nombre`) USING BTREE
)
COLLATE='utf8mb3_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=21
;
CREATE TABLE `carrito` (
	`fk_user_id` INT(10) NOT NULL,
	`fk_id_producto` INT(5) NOT NULL,
	INDEX `user_id` (`fk_user_id`) USING BTREE,
	INDEX `prod` (`fk_id_producto`) USING BTREE,
	CONSTRAINT `prod` FOREIGN KEY (`fk_id_producto`) REFERENCES `sounds`.`productos` (`id_producto`) ON UPDATE NO ACTION ON DELETE CASCADE,
	CONSTRAINT `user_id` FOREIGN KEY (`fk_user_id`) REFERENCES `sounds`.`usuarios` (`user_id`) ON UPDATE NO ACTION ON DELETE NO ACTION
)
COLLATE='utf8mb3_general_ci'
ENGINE=InnoDB
;
